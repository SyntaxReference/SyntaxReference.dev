```Go
// ------------------------------------
// Generic/Template
// ------------------------------------

// Since Go 1.18
func Identity[T any](value T) T {
	return value
}

func Contains[T comparable](values []T, target T) bool {
	for _, value := range values {
		if value == target {
			return true
		}
	}

	return false
}
```
