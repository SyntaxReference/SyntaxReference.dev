```Cs
// ------------------------------------
// Integer
// ------------------------------------
byte variableName = 10;		// Range: 0 to 255
sbyte variableName = 10;	// Range: -128 to 127
short variableName = 10;	// Range: -32,768 to 32,767
ushort variableName = 10;	// Range: 0 to 65,535
int variableName = 10;		// Range: -2,147,483,648 to 2,147,483,647
uint variableName = 10U;	// Range: 0 to 4,294,967,295
long variableName = 10L;
ulong variableName = 10UL;

// Const Declaration
const int MY_CONSTANT = 10;
readonly int myReadonlyValue = 10;

// ------------------------------------
// Float & Double
// ------------------------------------
float variableName = 10.5f;
double variableName = 10.5;
decimal variableName = 10.5m;

// Const Declaration
const double MY_CONSTANT = 10.5;

// ------------------------------------
// Character
// ------------------------------------
char variableName = 'a';
char variableName = '\n';

// ------------------------------------
// Boolean
// ------------------------------------
bool variableName = true;
bool variableName = false;
```
