```Java
// ------------------------------------
// Integer
// ------------------------------------
byte variableName = 10;		// Range: -128 to 127
short variableName = 10;	// Range: -32,768 to 32,767
int variableName = 10;		// Range: -2,147,483,648 to 2,147,483,647
long variableName = 10L;

// Const Declaration
final int MY_CONSTANT = 10;

// ------------------------------------
// Float & Double
// ------------------------------------
float variableName = 10.5f;
double variableName = 10.5;

// Const Declaration
final double MY_CONSTANT = 10.5;

// ------------------------------------
// Character
// ------------------------------------
char variableName = 'a';
char variableName = '\n';

// ------------------------------------
// Boolean
// ------------------------------------
boolean variableName = true;
boolean variableName = false;

// ------------------------------------
// Type Inference
// ------------------------------------
// Since Java 10, local variables can use "var".
var inferredInteger = 10;
```
