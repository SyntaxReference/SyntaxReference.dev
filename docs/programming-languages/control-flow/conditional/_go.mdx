```Go
// ------------------------------------
// If-Else
// ------------------------------------
if condition {
	// ...
} else if otherCondition {
	// ...
} else {
	// ...
}

// Short statement before condition
if value := getValue(); value > 10 {
	// ...
}
```

```Go
// ------------------------------------
// Ternary
// ------------------------------------
// Not Available.
// Use if/else.
```

```Go
// ------------------------------------
// Switch Case Statement
// ------------------------------------
switch value {
case 1:
	// ...
case 2, 3:
	// ...
default:
	// ...
}

// Switch without expression
switch {
case value < 0:
	// ...
case value == 0:
	// ...
default:
	// ...
}
```
