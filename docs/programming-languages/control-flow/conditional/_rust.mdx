```Rust
// ------------------------------------
// If-Else
// ------------------------------------
if condition {
	// ...
} else if other_condition {
	// ...
} else {
	// ...
}
```

```Rust
// ------------------------------------
// If as Expression
// ------------------------------------
let value = if condition {
	10
} else {
	20
};
```

```Rust
// ------------------------------------
// Match
// ------------------------------------
match value {
	1 => println!("One"),
	2 | 3 => println!("Two or Three"),
	4..=10 => println!("Four to Ten"),
	_ => println!("Something else"),
}

// Match as Expression
let result = match value {
	1 => "One",
	2 => "Two",
	_ => "Other",
};
```
