public struct Vector2
{
	public float X { get; }
	public float Y { get; }

	public Vector2(float x, float y)
	{
		X = x;
		Y = y;
	}

	public static Vector2 operator +(Vector2 first, Vector2 second)
	{
		return new Vector2(first.X + second.X, first.Y + second.Y);
	}
}

Vector2 result = first + second;
